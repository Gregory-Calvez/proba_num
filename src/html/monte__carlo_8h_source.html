126 template<typename Generator> void monte_carlo<Generator>::compute(Generator & gen){
128  double sum = 0.;
129  double sum_square = 0.;
130  double temp = 0.;
131  double confidence = 0.;
132  do{
133  ++count;
134  temp = (*variable)(gen);
135  sum += temp;
136  sum_square += std::pow(temp, 2);
137  confidence = 2*1.96*std::sqrt(sum_square/count - std::pow((sum/count), 2)) / std::sqrt(count);
138  } while(count < 100 || (confidence > precision && count < cap_iterations));
139  empirical_mean = sum / count;
140  empirical_var = sum_square / count - std::pow(empirical_mean, 2);
141  empirical_std = std::sqrt(empirical_var);
142  ci_l_bound = empirical_mean - 1.96 * empirical_std / std::sqrt(count);
143  ci_h_bound = empirical_mean + 1.96 * empirical_std / std::sqrt(count);
144  successful = (confidence < precision);
145  computed = true;
146 };
