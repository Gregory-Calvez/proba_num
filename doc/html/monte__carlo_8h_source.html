133 template<typename Generator> void monte_carlo<Generator>::compute(Generator & gen){
135 
136  // We use a chrono to measure the time taken to compute the estimation.
137  std::chrono::steady_clock::time_point begin = std::chrono::steady_clock::now();
138 
139  double sum = 0.;
140  double sum_square = 0.;
141  double temp = 0.;
142  double confidence = 0.;
143  do{
144  ++count;
145  temp = (*variable)(gen);
146  sum += temp;
147  sum_square += std::pow(temp, 2);
148  confidence = 2*1.96*std::sqrt(sum_square/count - std::pow((sum/count), 2)) / std::sqrt(count);
149  } while(count < 100 || (confidence > precision && count < cap_iterations));
150  std::chrono::steady_clock::time_point end= std::chrono::steady_clock::now();
151  this->empirical_mean = sum / count;
152  this->empirical_var = sum_square / count - std::pow(empirical_mean, 2);
153  this->empirical_std = std::sqrt(empirical_var);
154  this->ci_l_bound = empirical_mean - 1.96 * empirical_std / std::sqrt(count);
155  this->ci_h_bound = empirical_mean + 1.96 * empirical_std / std::sqrt(count);
156  this->successful = (confidence < precision);
157  this->computed = true;
158  this->elapsed_time_min = std::chrono::duration_cast<std::chrono::minutes>(end - begin).count();
159  this->elapsed_time_sec = std::chrono::duration_cast<std::chrono::seconds>(end - begin).count() - this->elapsed_time_min * 60;

186 template<typename Generator> void monte_carlo<Generator>::compute_control_variate(Generator & gen)
187 {
188  std::chrono::steady_clock::time_point begin = std::chrono::steady_clock::now();
189  if (this->variable->has_control() == false){
190  std::cout << "This variable has no control variate defined. " << std::endl;
191  return;
192  };
193 
194  // First, compute the control constant
195  double mean_control_variate = this->variable->get_mean_control_variate();
196  unsigned int num_iterations_to_compute_constant = std::floor(this->cap_iterations / 10);
197 
198  double sum_variable = 0;
199  double sum_control = 0;
200  double sum_covariance = 0;
201  double sum_order_2_control = 0;
202  std::pair<double, double> temp;
203  for(unsigned int i = 0; i<num_iterations_to_compute_constant; ++i){
204  temp = this->variable->simulate(gen);
205  sum_variable += temp.first;
206  sum_control += temp.second;
207  sum_covariance += temp.first*temp.second;
208  sum_order_2_control += std::pow(temp.second, 2);
209  };
210  double covariance = (sum_covariance / num_iterations_to_compute_constant) - mean_control_variate * (sum_variable / num_iterations_to_compute_constant);
211  double variance_control = (sum_order_2_control / num_iterations_to_compute_constant) - std::pow(mean_control_variate, 2);
212 
213  double c = - covariance / variance_control;
214 
215 
216  // Then, we do the monte carlo
217  double sum_mc = 0.;
218  double sum_mc_square = 0.;
219  double confidence = 0.;
220  double temp_x = 0;
221  unsigned int count = 0;
222  do{
223  ++count;
224  temp = this->variable->simulate(gen);
225  temp_x = temp.first + c * (temp.second - mean_control_variate);
226  sum_mc += temp_x;
227  sum_mc_square += std::pow(temp_x, 2);
228  confidence = 2*1.96*std::sqrt(sum_mc_square/count - std::pow((sum_mc/count), 2)) / std::sqrt(count);
229  } while(count < 100 || (confidence > precision && count < cap_iterations));
230  std::chrono::steady_clock::time_point end= std::chrono::steady_clock::now();
231  empirical_mean = sum_mc / count;
232  empirical_var = sum_mc_square / count - std::pow(empirical_mean, 2);
233  empirical_std = std::sqrt(empirical_var);
234  ci_l_bound = empirical_mean - 1.96 * empirical_std / std::sqrt(count);
235  ci_h_bound = empirical_mean + 1.96 * empirical_std / std::sqrt(count);
236  successful = (confidence < precision);
237  computed = true;
238  this->elapsed_time_min = std::chrono::duration_cast<std::chrono::minutes>(end - begin).count();
239  this->elapsed_time_sec = std::chrono::duration_cast<std::chrono::seconds>(end - begin).count() - this->elapsed_time_min * 60;
240  return ;
